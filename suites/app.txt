function parseFullImageToParts(fullImage) {
    var s = (fullImage || '').trim();
    if (!s) return { reg: '', img: '', tag: '' };
    var i = s.indexOf('/');
    if (i === -1) return { reg: s, img: '', tag: '' };

function parseDumpInBrowser(dump) {
    var out = {};
    var text = (dump || '').trim();
    if (!text) return out;
    var imageRe = /[a-zA-Z0-9][a-zA-Z0-9._-]*\/[a-zA-Z0-9\/._-]+:[a-zA-Z0-9._-]+/;
    var imageReLoose = /\S+\/\S+:\S+/;
    var lines = text.split(/\r?\n/);
    for (var i = 0; i < lines.length; i++) {
        var line = lines[i].trim();
        if (!line) continue;
        if (line.indexOf('://') !== -1 && line.indexOf('.repo') !== -1) continue;
        var m = line.match(imageRe) || line.match(imageReLoose);
        if (!m || !m[0]) continue;
        var img = m[0].trim();
        var low = img.toLowerCase();
        if (low.indexOf('nvmeof-cli') !== -1 || low.indexOf('nvmeof_cli') !== -1) {
            out.nvmeof_cli_image = img;
        } else if (low.indexOf('nvmeof') !== -1 && low.indexOf('cli') === -1) {
            out.nvmeof_image = img;
        } else if ((low.indexOf('ceph-9') !== -1 || low.indexOf('ceph-8') !== -1) && low.indexOf('crimson') === -1) {
            out.full_image = img;
        } else if (low.indexOf('ceph') !== -1 && low.indexOf('rhel') !== -1 && low.indexOf('crimson') === -1 && !out.full_image) {
            out.full_image = img;
        }
    }
    if (out.full_image) {
        var parts = parseFullImageToParts(out.full_image);
        if (parts.reg) out.docker_registry = parts.reg;
        if (parts.img) out.docker_image = parts.img;
        if (parts.tag) out.docker_tag = parts.tag;
    }
    var repoUrlMatch = text.match(/https?:\/\/[^\s<>'"]+\.repo/);
    if (repoUrlMatch) {
        out.rhs_repo = repoUrlMatch[0].trim();
    } else {
        var baseUrlMatch = text.match(/https?:\/\/[^\s<>'")]+/);
        if (baseUrlMatch) {
            var baseUrl = baseUrlMatch[0].trim();
            if (baseUrl.toLowerCase().indexOf('ceph') !== -1 || baseUrl.toLowerCase().indexOf('ibm') !== -1) {
                baseUrl = baseUrl.replace(/\/+$/, '');
                var segs = baseUrl.split('/');
                var lastSeg = segs[segs.length - 1] || '';
                if (lastSeg && lastSeg.indexOf('.') !== -1) {
                    out.rhs_repo = baseUrl + '/' + lastSeg + '-rhel10.repo';
                    out.rhs_repo_rhel9 = baseUrl + '/' + lastSeg + '-rhel9.repo';
                }
            }
        }
    }
    return out;
}

def full(p):
    return os.path.expanduser(p) if p else ""


def parse_full_image(full_image):
    """Parse full image (e.g. cp.stg.icr.io/cp/ibm-ceph/ceph-9-rhel9:v9.1-12786) into registry, image, tag."""
    s = (full_image or "").strip()
    if not s:
        return "", "", ""
    i = s.find("/")
    if i == -1:
        return s, "", ""
    registry = s[:i]
    rest = s[i + 1 :]
    j = rest.rfind(":")
    if j == -1:
        return registry, rest, ""
    return registry, rest[:j], rest[j + 1 :]

def list_dirs(base):
    
    if not os.path.exists(base): return []
    return [{"name":d,"path":os.path.join(base,d)} for d in os.listdir(base) if os.path.isdir(os.path.join(base,d))]

def list_yaml(folder):
    if not os.path.exists(folder):
        return []
    return [{"name":f,"path":os.path.join(folder,f)} for f in os.listdir(folder) if f.endswith(('.yaml','.yml'))]

        for k, v in data.items():
            key = (prefix + "." + k if prefix else k).lower().replace("-", "_")
            if key in ("ceph_base", "ceph-base") and v:
                out["full_image"] = _normalize_image(v)
            elif key in ("nvmeof_cli_image", "nvmeof-cli-image") and v:
                out["nvmeof_cli_image"] = _normalize_image(v)
            elif key in ("nvmeof_image", "nvmeof-image") and v:
                out["nvmeof_image"] = _normalize_image(v)
            elif key == "default" and isinstance(v, dict) and "rhel" in str(v).lower():
                for rhel_k, repo_url in v.items():
                    if isinstance(repo_url, str) and ".repo" in repo_url:
                        out.setdefault("rhs_repo", repo_url)
            elif isinstance(v, str) and ".repo" in v and ("repos" in key or "repo" in key or "rhel" in key):
                out["rhs_repo"] = _normalize_image(v)
            if isinstance(v, (dict, list)):
                _extract_from_yaml(v, out, key)
    elif isinstance(data, list):
        for item in data:
            _extract_from_yaml(item, out, prefix)


# Image: registry/path:tag - permissive (tag can contain digits, dots, hyphens)
_IMAGE_RE = re.compile(
    r"(?P<image>[a-zA-Z0-9][a-zA-Z0-9._-]*/[a-zA-Z0-9/._-]+:[a-zA-Z0-9./_-]+)"
)

# Any HTTP URL (candidate repo directory; may end with / or a segment like .ci.0)
_REPO_BASE_URL_RE = re.compile(r"(https?://[^\s<>'\")\]]+)")

# Direct .repo URL
_REPO_URL_RE = re.compile(r"https?://[^\s<>'\"]+\.repo")

# key: value or key=value (value = rest of line or single token)
_KV_LINE_RE = re.compile(
    r"^\s*(?P<key>ceph-base|ceph_base|full_image|nvmeof_cli_image|nvmeof_image|rhs.repo|rhs_ceph_repo)"
    r"\s*[:=]\s*(?P<val>.+)$",
    re.IGNORECASE,
)


def parse_dump(dump_text):
    """Extract full_image, nvmeof_cli_image, nvmeof_image, rhs_repo from pasted dump."""
    out = {}
    text = (dump_text or "").strip()
    if not text:
        return out

    # 0) Line-by-line container list (e.g. "  cp.stg.icr.io/cp/ibm-ceph/ceph-9-rhel9:v9.1-12786")
    for line in text.splitlines():
        line = line.strip()
        if not line or "://" in line and ".repo" in line:
            continue
        m = _IMAGE_RE.search(line)
        if not m:
            continue
        img = m.group("image").strip()
        if not img or img.count("/") < 1 or ":" not in img:
            continue
        low = img.lower()
        if "nvmeof-cli" in low or "nvmeof_cli" in low:
            out["nvmeof_cli_image"] = img
        elif "nvmeof" in low and "cli" not in low:
            out["nvmeof_image"] = img
        elif ("ceph-9" in low or "ceph-8" in low) and "crimson" not in low:
            out["full_image"] = img
        elif "ceph" in low and "rhel" in low and "crimson" not in low and "full_image" not in out:
            out["full_image"] = img

    # 1) Try YAML (full or partial) - only if we didn't get images from line-by-line
    if not out:
        try:
            data = yaml.safe_load(text)
            if data:
                _extract_from_yaml(data, out)
        except Exception:
            pass

    # 2) Key: value / key=value (line-by-line)
    for line in text.splitlines():
        m = _KV_LINE_RE.match(line.strip())
        if not m:
            continue
        key = m.group("key").lower().replace("-", "_")
        val = m.group("val").strip().strip("'\"").split()[0] if m.group("val") else ""
        val = _normalize_image(val)
        if not val:
            continue
        if "ceph" in key or key == "full_image":
            out["full_image"] = val
        elif "nvmeof_cli" in key:
            out["nvmeof_cli_image"] = val
        elif "nvmeof_image" in key:
            out["nvmeof_image"] = val
        elif "repo" in key:
            out["rhs_repo"] = val

    # 3) Loose key: value (common keys)
    for line in text.splitlines():
        line = line.strip()
        for sep in (":", "="):
            if sep not in line:
                continue
            idx = line.index(sep)
            k = line[:idx].strip().lower().replace("-", "_")
            v = line[idx + 1 :].strip().strip("'\"").split()[0]
            if not v or "://" in k:
                continue
            if k in ("ceph_base", "ceph-base", "full_image") and "/" in v and ":" in v:
                out["full_image"] = _normalize_image(v)
            elif k in ("nvmeof_cli_image", "nvmeof_cli") and "/" in v:
                out["nvmeof_cli_image"] = _normalize_image(v)
            elif k in ("nvmeof_image", "nvmeof") and "/" in v and "cli" not in v.lower():
                out["nvmeof_image"] = _normalize_image(v)
            break

    # 4) Direct .repo URLs
    for m in _REPO_URL_RE.finditer(text):
        out["rhs_repo"] = m.group(0).strip()

    # 4b) Repo directory base URL (e.g. "RHEL repos here: https://.../IBM-CEPH-9.1-202603162056.ci.0/")
    #     Construct .repo links: base + last_segment + "-rhel9.repo" / "-rhel10.repo"
    if "rhs_repo" not in out or not out.get("rhs_repo"):
        for m in _REPO_BASE_URL_RE.finditer(text):
            base_url = m.group(1).strip()
            if "ceph" not in base_url.lower() and "ibm" not in base_url.lower():
                continue
            base_url = base_url.rstrip("/")
            last_segment = base_url.split("/")[-1] if "/" in base_url else ""
            if not last_segment or "." not in last_segment:
                continue
            base_url += "/"
            out["rhs_repo"] = base_url + last_segment + "-rhel10.repo"
            out["rhs_repo_rhel9"] = base_url + last_segment + "-rhel9.repo"
            break

    # 5) Standalone image (fallback: find anywhere in text)
    for m in _IMAGE_RE.finditer(text):
        img = m.group("image").strip()
        if not img or img.count("/") < 1 or ":" not in img:
            continue
        low = img.lower()
        if "nvmeof-cli" in low or "nvmeof_cli" in low:
            out.setdefault("nvmeof_cli_image", img)
        elif "nvmeof" in low and "cli" not in low:
            out.setdefault("nvmeof_image", img)
        elif ("ceph-9" in low or "ceph-8" in low or "ceph_base" in low or ("ceph" in low and "rhel" in low)) and "crimson" not in low:
            out.setdefault("full_image", img)

    # Ensure docker fields from full_image
    if out.get("full_image"):
        reg, img, tag = parse_full_image(out["full_image"])
        if reg:
            out["docker_registry"] = reg
        if img:
            out["docker_image"] = img
        if tag:
            out["docker_tag"] = tag

    return out

def _test_name(entry):
    """Extract display name from a suite test entry (supports nested test: { name: ... })."""
    if not isinstance(entry, dict):
        return str(entry)
    inner = entry.get('test') or entry
    if not isinstance(inner, dict):
        return str(entry)
    return (
        inner.get('name') or
        inner.get('test') or
        inner.get('desc') or
        str(entry)
    )

# ---------- Manifest: fetch RHS Repo (and NVMe images) from qe-ceph-manifest ----------
MANIFEST_BASE = "https://raw.githubusercontent.com/ibmstorage/qe-ceph-manifest/main/ibm"

def _fetch_manifest(version):
    url = f"{MANIFEST_BASE}/{version}.yaml"
    try:
        req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0 (compatible; CephCI/1.0)"})
        with urllib.request.urlopen(req, timeout=15) as r:
            return yaml.safe_load(r.read())
    except Exception:
        return None

def _safe_image(val):
    if val is None or (isinstance(val, str) and val.strip().lower() == "none"):
        return ""
    return (val or "").strip()

@app.route('/manifest/<version>')
def manifest(version):
    rhel = request.args.get('rhel', 'rhel-9')
    if version not in ('9.0', '9.1'):
        return jsonify({})
    data = _fetch_manifest(version)
    if not data:
        return jsonify({"error": "Could not fetch manifest"})
    # Use stable if present, else nightly (from default section for repo)
    channel = "stable" if "stable" in data else "nightly"
    channel_data = data.get(channel) or {}
    repos = channel_data.get("repositories", {}).get("default", {})
    rhs_repo = (repos or {}).get(rhel) or ""
    images = channel_data.get("images") or {}
    return jsonify({
        "rhs_repo": rhs_repo,
        "full_image": _safe_image(images.get("ceph-base")),
        "nvmeof_cli_image": _safe_image(images.get("nvmeof_cli_image")),
        "nvmeof_image": _safe_image(images.get("nvmeof_image")),
    })

# ---------- TEMP SUITE ----------
last_preview_temp = None  # cleaned up when a new preview is created

def create_temp_suite(orig_file, selected_names):
    selected = set(selected_names or [])
    with open(full(orig_file)) as f:
        data = yaml.safe_load(f)
    if not data:
        data = {'tests': []}
    filtered = []
    for t in data.get('tests', []):
        if _test_name(t) in selected:
            filtered.append(t)
    data['tests'] = filtered

    # Temp name resembles selected suite: e.g. tier-2_nvmeof_stretch_cluster_subset_a1b2c3d4.yaml
    base = os.path.splitext(os.path.basename(orig_file))[0]
    safe_base = "".join(c if c.isalnum() or c in "-_" else "_" for c in base)
    short_id = uuid.uuid4().hex[:8]
    tmp = os.path.join(tempfile.gettempdir(), f"{safe_base}_subset_{short_id}.yaml")

    with open(tmp, 'w') as f:
        yaml.dump(data, f)

    return tmp

def _rel_path(path, root):
    """Return path relative to root if under root, else absolute."""
    if not path:
        return path
    path = full(path)
    root = os.path.normpath(root)
    try:
        rel = os.path.relpath(path, root)
        if not rel.startswith(".."):
            return rel
    except ValueError:
        pass
    return path


def _build_command_args(d, temp_suite_path, root):
    """Build argv list for run.py (without creating temp suite)."""
    global_conf = full(d.get("global_conf") or "")
    inventory = full(d.get("inventory") or "")
    reuse_file = full(d.get("reuse_file") or "") if d.get("reuse_file") else ""
    osp_cred = full(d.get("osp_cred") or "")
    cmd = ["python3", "run.py"]
    if d.get("log_level"):
        cmd += ["--log-level", d["log_level"]]
    if d.get("cloud"):
        cmd += ["--cloud", d["cloud"]]
    if d.get("instances_name"):
        cmd += ["--instances-name", d["instances_name"]]
    if d.get("rhbuild"):
        cmd += ["--rhbuild", d["rhbuild"]]
    if d.get("platform"):
        cmd += ["--platform", d["platform"]]
    cmd += [
        "--global-conf", _rel_path(d.get("global_conf"), root) or global_conf,
        "--suite", temp_suite_path,
        "--inventory", _rel_path(d.get("inventory"), root) or inventory,
    ]
    if osp_cred:
        cmd += ["--osp-cred", osp_cred]
    if d.get("nvmeof_image"):
        cmd += ["--custom-config", "nvmeof_image=" + d["nvmeof_image"]]
    if d.get("nvmeof_cli_image"):
        cmd += ["--custom-config", "nvmeof_cli_image=" + d["nvmeof_cli_image"]]
    if d.get("rhs_repo"):
        cmd += ["--rhs-ceph-repo", d["rhs_repo"]]
    reg, img, tag = parse_full_image(d.get("full_image") or "")
    reg = reg or d.get("docker_registry") or ""
    img = img or d.get("docker_image") or ""
    tag = tag or d.get("docker_tag") or ""
    if reg:
        cmd += ["--docker-registry", reg]
    if img:
        cmd += ["--docker-image", img]
    if tag:
        cmd += ["--docker-tag", tag]
    for line in (d.get("extra_custom_config") or "").strip().splitlines():
        line = line.strip()
        if line and "=" in line:
            cmd += ["--custom-config", line]
    if d.get("skip_subscription"):
        cmd.append("--skip-subscription")
    if d.get("mode") == "new":
        cmd.append("--store")
    elif reuse_file:
        cmd += ["--reuse", reuse_file]
    return cmd


def build_command(d):
    suite_path = d.get("suite") or ""
    tests = d.get("tests") or []
    if not suite_path:
        raise ValueError("Suite file is required")
    temp_suite = create_temp_suite(suite_path, tests)
    root = full(d.get("root") or os.getcwd())
    cmd = _build_command_args(d, temp_suite, root)
    return cmd, root, temp_suite

def _cmd_to_string(cmd):
    """Format argv list as a single shell command string (quote args with spaces)."""
    return " ".join(repr(a) if " " in a else a for a in cmd)


@app.route('/preview', methods=['POST'])
def preview():
    global last_preview_temp
    data = request.json or {}
    suite_path = data.get("suite") or ""
    tests = data.get("tests") or []
    if not suite_path:
        return jsonify({"error": "Suite file is required"})
    try:
        if last_preview_temp and os.path.isfile(last_preview_temp):
            try:
                os.remove(last_preview_temp)
            except OSError:
                pass
        temp_suite = create_temp_suite(suite_path, tests)
        last_preview_temp = temp_suite
    except Exception as e:
        return jsonify({"error": str(e)})
    root = full(data.get("root") or os.getcwd())
    cmd = _build_command_args(data, temp_suite, root)
    return jsonify({"command": _cmd_to_string(cmd), "temp_suite_path": temp_suite})

@app.route('/run', methods=['POST'])
def run():
    data = request.json or {}
    try:
        cmd, cwd, temp_suite_path = build_command(data)
    except (KeyError, TypeError, ValueError) as e:
        return jsonify({"ok": False, "error": str(e)}), 400
    threading.Thread(target=exec_cmd, args=(cmd, cwd, temp_suite_path)).start()
    return jsonify({"ok": True})

def exec_cmd(cmd, cwd=None, temp_suite_path=None):
    global logs
    logs.clear()
    try:
        p = subprocess.Popen(
            cmd,
            stdout=subprocess.PIPE,
            stderr=subprocess.STDOUT,
            text=True,
            cwd=cwd or os.getcwd(),
        )
        for line in p.stdout:
            logs.append(line)
        p.wait()
    finally:
        if temp_suite_path and os.path.isfile(temp_suite_path):
            try:
                os.remove(temp_suite_path)
            except OSError:
                pass

@app.route('/logs')
def get_logs():
    return jsonify(logs)

if __name__ == "__main__":
    app.run(debug=True)
